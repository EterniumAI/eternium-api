Quick Start

import { Eternium } from 'eternium-sdk'

const client = new Eternium('YOUR_KEY')

// Generate image
const img = await client.image('A mountain landscape, golden hour')
console.log(img.url)

// Generate video
const vid = await client.video('Drone flyover of a city at night')
console.log(vid.url)

// Run a pipeline
const pack = await client.runPipeline('product-shot', 'White sneakers')
pack.results.forEach(r => console.log(r.url))
from eternium import Eternium

client = Eternium("YOUR_KEY")

img = client.image("A mountain landscape, golden hour")
print(img["url"])

vid = client.video("Drone flyover of a city at night")
print(vid["url"])

pack = client.run_pipeline("product-shot", "White sneakers")
for r in pack["results"]:
    print(r["url"])

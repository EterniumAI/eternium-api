What the SDK handles

Task submission and automatic polling until completion. Configurable timeouts (default 5 min). Progress callbacks for UI updates. Automatic download URL retrieval. Pipeline orchestration with parallel task waiting. Prompt caching with $0 cost on cache hits.

import { Eternium } from 'eternium-sdk'

const client = new Eternium('etrn_your_key')

// Generate image — returns download URL
const img = await client.image('A futuristic city at sunset')
console.log(img.url)

// Generate video with options
const vid = await client.video('Drone flyover at night', {
  duration: 10, sound: true, mode: 'pro'
})
console.log(vid.url)

// Run a pipeline — 3 product shots, one call
const pack = await client.runPipeline(
  'product-shot',
  'White sneakers on marble'
)
pack.results.forEach(r => console.log(r.url))

// Check your spend
const usage = await client.getUsage()
console.log(`${usage.remaining} remaining`)
from eternium import Eternium

client = Eternium("etrn_your_key")

# Generate image — returns download URL
img = client.image("A futuristic city at sunset")
print(img["url"])

# Generate video with options
vid = client.video("Drone flyover at night",
  duration=10, sound=True, mode="pro"
)
print(vid["url"])

# Run a pipeline — 3 product shots, one call
pack = client.run_pipeline(
  "product-shot",
  "White sneakers on marble"
)
for r in pack["results"]:
    print(r["url"])

# Check your spend
usage = client.get_usage()
print(f"{usage['remaining']} remaining")
# Generate an image
curl -X POST https://api.eternium.ai/v1/generate \
  -H "Content-Type: application/json" \
  -H "X-API-Key: etrn_your_key" \
  -d '{"model":"nano-banana-pro",
       "prompt":"A futuristic city at sunset"}'

# Run a pipeline
curl -X POST https://api.eternium.ai/v1/pipelines/run \
  -H "Content-Type: application/json" \
  -H "X-API-Key: etrn_your_key" \
  -d '{"pipeline":"product-shot",
       "prompt":"White sneakers on marble"}'

# Check task status
curl https://api.eternium.ai/v1/tasks/TASK_ID \
  -H "X-API-Key: etrn_your_key"

# Check your usage
curl https://api.eternium.ai/v1/usage \
  -H "X-API-Key: etrn_your_key"
